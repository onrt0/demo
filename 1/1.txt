git clone [address]
git init #创建本地代码仓库
git remote add origin [address] #连接远程库和本地库
git add a.txt #放入暂存区
git add . #所有文件都放入暂存区
git commit -m "备注信息"
git log
git log --stat #查看每次提交修改了哪些文件
git diff [commit id] #查看某次提交修改了什么内容
git reset --hard [commit id] #永久性回退
git checkout [commit id] #临时跳转查看，若在此提交新代码，需合并或创建分支否则丢失
git branch #查看所有分支
git checkout -b develop #创建develop分支，并切换到develop分支
###合并分支
git add .
git commit -m "备注信息"
git checkout master #切换到master分支
git merge develop
git push origin main #将先前的本地库origin更新到GitHub上的main分支中
# 删除多余的远程别名
git remote remove demo
# 确认清理结果
git remote -v
git remote set-url new [new address] #修改新地址
git pull origin main #若无相关历史，则git pull origin main --allow-unrelated-histories
git log --oneline #查看本地提交记录
---
本地分支名要和默认分支相同，不然直接使用git pull origin main报错
远程仓库为main分支，本地分支名也需要修改为main：
git branch -m master main #修改分支名
或者push时加上：
git push origin master:main #本地为master分支，远程为main分支
